root@forge:~# cat /etc/nginx/sites-available/firesky
server {
    server_name fireskyops.tech www.fireskyops.tech;

    location /api/ {
        proxy_pass http://localhost:8080/api/;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_read_timeout 300s;
        proxy_cache_bypass $http_upgrade;
    }

    location /sky-vision/ {
        alias /var/www/firesky/artifacts/sky-vision/dist/public/;
        try_files $uri $uri/ /sky-vision/index.html;
    }

    location / {
        root /var/www/firesky/artifacts/firesky/dist/public;
        try_files $uri $uri/ /index.html;
    }

    listen 443 ssl; # managed by Certbot
    ssl_certificate /etc/letsencrypt/live/fireskyops.tech/fullchain.pem; # managed by Certbot
    ssl_certificate_key /etc/letsencrypt/live/fireskyops.tech/privkey.pem; # managed by Certbot
    include /etc/letsencrypt/options-ssl-nginx.conf; # managed by Certbot
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem; # managed by Certbot




}
server {
    if ($host = fireskyops.tech) {
        return 301 https://$host$request_uri;
    } # managed by Certbot


    listen 80;
    server_name fireskyops.tech www.fireskyops.tech;
    return 404; # managed by Certbot
